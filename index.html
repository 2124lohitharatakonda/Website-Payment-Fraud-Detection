🗄️ SQL — Fraud Query
-- High-risk transactions last 24h
SELECT session_id, amount,
  ip_country, fraud_score,
  signal_type
FROM transactions
WHERE created_at > NOW() - INTERVAL '24h'
  AND fraud_score > 0.85
  AND status = 'flagged'
ORDER BY fraud_score DESC
LIMIT 100;
